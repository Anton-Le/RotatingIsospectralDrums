158  }

359 void ref2d(geometry2& Pars, const uint Nref) {
360 
361  /* Vor dem Aufruf der Funktion Gebietsverfeinerung sollte geprueft werden,
362  ob Pars.Verfeinerung > 0 erfuellt ist. Falls nicht fuehrt dieses Programm
363  nicht zur gewuenschten Verfeinerung.
364  */
365  if (Nref == 0)
366  return;
367 
368  // ##############################################################
369  // Berechnen, wiegross die verfeinerten Strukturen werden:
370  uint vorfaktor, Anz_Elemente, Anz_Knoten, Anz_Kanten, Anz_Rand_D, Anz_Rand_N;
371  vorfaktor = pow(2, Nref);
372 
373  Anz_Elemente = vorfaktor * vorfaktor * Pars.Nelem;
374  Anz_Rand_D = vorfaktor * Pars.NBdElemD;
375  Anz_Rand_N = vorfaktor * Pars.NBdElemN; // Falls NeumannRand
376  Anz_Kanten = (3 * Anz_Elemente + Anz_Rand_D) / 2;
377  Anz_Kanten = (3 * Anz_Elemente + Anz_Rand_D + Anz_Rand_N) / 2; // Falls NeumannRand
378  Anz_Knoten = 1 + Anz_Kanten - Anz_Elemente;
379 
380  //allocate memory for the refined mesh
381  int **Elemente, **Randdaten_D, **Randdaten_N;
382  dtype **Koordinaten;
383 
384  Elemente = new int*[Anz_Elemente];
385  for (uint j(0); j < Anz_Elemente; ++j)
386  Elemente[j] = new int[3];
387 
388  Randdaten_D = new int*[Anz_Rand_D];
389  for (uint j(0); j < Anz_Rand_D; ++j)
390  Randdaten_D[j] = new int[2];
391

393  for (uint j(0); j < Anz_Rand_N; ++j)
394  Randdaten_N[j] = new int[2];

793
