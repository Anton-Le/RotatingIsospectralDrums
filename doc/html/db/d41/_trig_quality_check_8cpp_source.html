117  dtype meshSize(0);

122  geometry2 refGeo;

129  uint k(1), m(1);

133  for (uint ref(1); ref < maxRef; ++ref) {

215 dtype L2NormRHS( const geometry2& geo, const dtype GD[][3], const uint QuadOrder) {
216  dtype NrmSq(0), integral(0), jac(0);
217 
218  Eigen::Vector2d tmp;
219  Eigen::Matrix2d J;
220  //Sum over all elements
221  for (uint k(0); k < geo.Nelem; ++k) {
222  Jacobian(geo.element[k], geo.vertices, J);
223  jac = fabs(J(0, 0) * J(1, 1) - J(1, 0) * J(0, 1));
224 
225  //compute the integral over the element
226  integral = 0;
227  for (uint s(0); s < QuadOrder; ++s) {
228  //map the quadrature nodes from the std simplex to the triangle
229  tmp(0) = GD[s][0];
230  tmp(1) = GD[s][1];
231  g(geo.element[k], geo.vertices, tmp);
232  integral += GD[s][2] * SQU(f( tmp(0), tmp(1)));
233  }
234  integral *= jac;
235  //add to the total
236  NrmSq += integral;
237  }
238  //return the ||f||_L^2 = sqrt( ||f||^2 )
239  return std::sqrt(NrmSq);
240 }

242 void setUpRHS(const geometry2& geo, const dtype GD[][3], const uint QuadOrder, Vec& RHS) {
243 
244  dtype integral(0), jac(0);
245  uint FuncIdx1 = 0;
246 
247  Eigen::Vector2d tmp;
248  Eigen::Matrix2d J;
249 
250  for (uint k(0); k < geo.Nelem; ++k) { //iteration over the elements(subintervals)
251  //compute the volume of the element
252  Jacobian(geo.element[k], geo.vertices, J);
253  //abosulte value of the determinant
254  jac = fabs(J(0, 0) * J(1, 1) - J(1, 0) * J(0, 1));
255 
256  for (uint i(0); i < triVert; ++i) { //iteration over the boundaries
257 
258  //determine function index
259  FuncIdx1 = geo.fIdx[ geo.element[k].vert[i] ];
260  //INHOMOGENOUS DIRICHLET ADDITION
261  if (geo.isInDirichletBoundary[ geo.element[k].vert[i] ] == true) {
262  continue; //skip the rest
263  }
264 
265  //integrate
266  integral = 0.0;
267 
268  for (uint j(0); j < QuadOrder; ++j) {
269  //map to the appropriate element
270  tmp(0) = GD[j][0];
271  tmp(1) = GD[j][1];
272  g(geo.element[k], geo.vertices, tmp);
273  integral += GD[j][2] * f(tmp(0), tmp(1)) * Phi(i, GD[j][0], GD[j][1]);
274  }
275  integral *= jac; //transform the measure (constant)
276  RHS(FuncIdx1) += integral; //we skip first node
277 
278  }
279  }

282 dtype L2FuncErr( const Vec& vals, const geometry2& geo, const dtype GD[][3], const uint QuadOrder) {
283  dtype NrmSq(0), integral(0), jac(0), arg(0);
284  uint FuncIdx1 = 0;
285 
286  Eigen::Vector2d tmp;
287  Eigen::Matrix2d J;
288  //Sum over all elements
289  for (uint k(0); k < geo.Nelem; ++k) {
290  Jacobian(geo.element[k], geo.vertices, J);
291  jac = fabs(J(0, 0) * J(1, 1) - J(1, 0) * J(0, 1));
292 
293  //compute the integral over the element
294  integral = 0;
295  for (uint s(0); s < QuadOrder; ++s) {
296  //map the quadrature nodes from the std simplex to the triangle
297  tmp(0) = GD[s][0];
298  tmp(1) = GD[s][1];
299  g(geo.element[k], geo.vertices, tmp);
300 
301  //sum over the basis functions on the std-simplex
302  arg = 0;
303  for (uint j(0); j < triVert; ++j) {
304  //skip dirichlet boundary
305  if (geo.isInDirichletBoundary[ geo.element[k].vert[j] ] == true)
306  continue;
307  FuncIdx1 = geo.fIdx[ geo.element[k].vert[j] ];
308  arg += vals(FuncIdx1) * Phi(j, GD[s][0], GD[s][1]);
309  }
310  //switch the sign and add the analytic value
311  arg -= u( tmp(0), tmp(1));
312 
313 
314  integral += GD[s][2] * SQU(arg);
315  }
316  integral *= jac;
317  //add to the total
318  NrmSq += integral;
319  }
320  //return the ||f||_L^2 = sqrt( ||f||^2 )
321  return std::sqrt(NrmSq);
322 }

324 dtype L2DerivErr(const Vec& vals, const geometry2& geo, const dtype GD[][3], const uint QuadOrder) {
325  dtype NrmSq(0), integral(0), jac(0);
326  uint FuncIdx1 = 0;
327 
328  Eigen::Vector2d tmp, dPhi1, dPhi2, arg, tmp2;
329  Eigen::Matrix2d J, Jinv;
330  //Sum over all elements
331  for (uint k(0); k < geo.Nelem; ++k) {
332  Jacobian(geo.element[k], geo.vertices, J);
333  jac = fabs(J(0, 0) * J(1, 1) - J(1, 0) * J(0, 1));
334  InvJacobian(geo.element[k], geo.vertices, Jinv);
335 
336  //compute the integral over the element
337  integral = 0;
338  for (uint s(0); s < QuadOrder; ++s) {
339  //map the quadrature nodes from the std simplex to the triangle
340  tmp(0) = GD[s][0];
341  tmp(1) = GD[s][1];
342  g(geo.element[k], geo.vertices, tmp);
343 
344  //sum over the gradients of the basis functions
345  arg.setZero();
346  for (uint j(0); j < triVert; ++j) {
347  //skip dirichlet boundary
348  if (geo.isInDirichletBoundary[ geo.element[k].vert[j] ] == true)
349  continue;
350  tmp2(0) = GD[s][0];
351  tmp2(1) = GD[s][1];
352  dPhi(j, tmp2, GD[s][0], GD[s][1]);
353  dPhi1 = Jinv*tmp2;
354 
355  FuncIdx1 = geo.fIdx[ geo.element[k].vert[j] ];
356  arg += vals(FuncIdx1) * dPhi1;
357  }
358  arg *= -1;
359  du(tmp2, tmp(0), tmp(1));
360  //dPhi2 = Jinv*tmp2;
361  dPhi2 = tmp2;
362  arg += dPhi2;
363 
364  integral += GD[s][2] * SQU(arg.norm());
365  }
366  integral *= jac;
367  //add to the total
368  NrmSq += integral;
369  }
370  //return the ||f||_L^2 = sqrt( ||f||^2 )
371  return std::sqrt(NrmSq);
372 }
