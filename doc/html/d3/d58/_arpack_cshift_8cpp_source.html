35 int main(int argc, char** argv) {
36  std::cout << "%SUITE_STARTING% ArpackCshift" << std::endl;
37  std::cout << "%SUITE_STARTED%" << std::endl;
38 
39  uint dim(100);
40  //create matrices
41  Eigen::SparseMatrix<dtype> A(dim, dim);
42  Eigen::SparseMatrix<dtype> B(dim, dim);
43  Eigen::VectorXi NnzPerRowA(dim), NnzPerRowB(dim);
44 
45  //fill the number of non-zeros - overestimate a bit
46  for (uint row(0); row < dim; ++row) {
47  NnzPerRowA(row) = 3;
48  NnzPerRowB(row) = 3;
49  }
50 
51  //reserve space
52  A.reserve(NnzPerRowA);
53  B.reserve(NnzPerRowB);
54 
55  //fill
56  for (uint row(1); row < dim - 1; ++row) {
57  //matrix A: diagonal: 2, subdiagonal: -2, superdiagonal: 3
58  A.coeffRef(row, row) = 2;
59  A.coeffRef(row, row - 1) = -2;
60  A.coeffRef(row, row + 1) = 3;
61  //matrix B: diagonal : 4 off-diagonal: 1
62  B.coeffRef(row, row) = 4;
63  B.coeffRef(row, row - 1) = B.coeffRef(row, row + 1) = 1;
64  }
65  //the first and last row
66  A.coeffRef(0, 0) = 2;
67  A.coeffRef(0, 1) = 3;
68  A.coeffRef(dim - 1, dim - 1) = 2;
69  A.coeffRef(dim - 1, dim - 2) = -2;
70  B.coeffRef(0, 0) = 4;
71  B.coeffRef(0, 1) = 1;
72  B.coeffRef(dim - 1, dim - 1) = 4;
73  B.coeffRef(dim - 1, dim - 2) = 1;
74 
75 
76  //compress
77  A.makeCompressed();
78  B.makeCompressed();
79 
80  //misc vectors
81  Vec w(dim), z(dim);
82  cVec wC(dim), zC(dim);
83 
84  //solver
85  cdtype ComplexShift(0.4, 0.6);
86  Eigen::UmfPackLU < Eigen::SparseMatrix<cdtype> > Dsolver;
87  Eigen::SparseMatrix<cdtype> D(dim, dim);
88  D = B.cast<cdtype>();
89  D *= -ComplexShift;
90  D += A.cast<cdtype>();
91  D.makeCompressed();
92 
93  Dsolver.compute(D);
94  //set up arpack
95  ARrcNonSymGenEig<dtype> prob(A.cols(), 4L, 'R', ComplexShift.real(), ComplexShift.imag());
96 
97  while (!prob.ArnoldiBasisFound()) {
98 
99  // Calling ARPACK FORTRAN code. Almost all work needed to
100  // find an Arnoldi basis is performed by TakeStep.
101 
102  prob.TakeStep();
103 
104  switch (prob.GetIdo()) {
105  case -1:
106 
107  // Performing w <- Re{OP*B*v} for the first time.
108  // This product must be performed only if GetIdo is equal to
109  // -1. GetVector supplies a pointer to the input vector, v,
110  // and PutVector a pointer to the output vector, w.
111  z = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(prob.GetVector(), z.rows());
112  w = B*z;
113  wC = w.cast<cdtype>();
114  //solve
115  zC = Dsolver.solve(wC);
116  for (uint row(0); row < zC.rows(); ++row)
117  z(row) = zC(row).real();
118  Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(prob.PutVector(), z.rows()) = z;
119  break;
120 
121  case 1:
122 
123  // Performing w <- Real{OP*B*v} when Bv is available.
124  // This product must be performed whenever GetIdo is equal to
125  // 1. GetProd supplies a pointer to the previously calculated
126  // product Bv and PutVector a pointer to the output vector w.
127 
128  z = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(prob.GetProd(), z.rows());
129  wC = z.cast<cdtype>();
130  //solve
131  zC = Dsolver.solve(wC);
132  for (uint row(0); row < zC.rows(); ++row)
133  w(row) = zC(row).real();
134  Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(prob.PutVector(), w.rows()) = w;
135  break;
136 
137  case 2:
138 
139  // Performing w <- B*v.
140  z = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(prob.GetVector(), z.rows());
141  w = B*z;
142  Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(prob.PutVector(), w.rows()) = w;
143 
144 
145  }
146  }
147 
148  // Finding eigenvalues and eigenvectors.
149 
150  int nconv = prob.FindEigenvectors();
151 
152  // Recovering eigenvalues of the original problem
153  // using the Rayleigh quotient.
154 
155  dtype num(0), denom(0), *evR, *evI, *rawEV;
156  cdtype numC(0), denomC(0), quotient(0);
157  uint ColJ(0), ColJp1(0);
158  evR = prob.RawEigenvalues();
159  evI = prob.RawEigenvaluesImag();
160  rawEV = prob.RawEigenvectors();
161  for (int i(0); i < nconv; ++i) {
162  ColJ = i * A.cols();
163  ColJp1 = ColJ + A.cols();
164  if (evI[i] == 0.0) {
165  //real eigenvalue
166  z = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(&rawEV[ColJ], z.rows());
167  w = A * z;
168  num = w.dot(z);
169  w = B * z;
170  denom = w.dot(z);
171  //store
172  evR[i] = num / denom;
173  std::cout << "lambda[" << i + 1 << "]: " << std::setprecision(10) << evR[i] << std::endl;
174  }
175  else {
176  //complex eigenvalue
177  //numerator
178  z = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(&rawEV[ColJ], z.rows());
179  w = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(&rawEV[ColJp1], w.rows());
180  for (uint s(0); s < w.rows(); ++s)
181  zC(s) = cdtype(z(s), w(s));
182  wC = A.cast<cdtype>() * zC;
183  numC = wC.dot(zC);
184 
185 
186 
187  //denominator
188  z = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(&rawEV[ColJ], z.rows());
189  w = Eigen::Matrix<dtype, Eigen::Dynamic, 1>::Map(&rawEV[ColJp1], w.rows());
190  for (uint s(0); s < w.rows(); ++s)
191  zC(s) = cdtype(z(s), w(s));
192  wC = B.cast<cdtype>() * zC;
193  denomC = wC.dot(zC);
194 
195  quotient = numC / denomC;
196  evR[i] = quotient.real();
197  evI[i] = quotient.imag();
198 
199  evR[i + 1] = std::conj(quotient).real();
200  evI[i + 1] = std::conj(quotient).imag();
201  i++;
202 
203  std::cout << "lambda[" << i + 1 << "]: " << std::setprecision(10) << " (" << evR[i] << ", " << evI[i] << " )" << std::endl;
204  std::cout << "lambda[" << i + 2 << "]: " << std::setprecision(10) << " (" << evR[i + 1] << ", " << evI[i + 1] << " )" << std::endl;
205 
206  }
207  }
